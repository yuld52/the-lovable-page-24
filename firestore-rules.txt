// Firebase Firestore Rules - Allow server to write

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Settings
    match /settings/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && request.auth.uid == userId || request.auth == null;
    }
    
    // Products - server can write without auth
    match /products/{productId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null || request.auth == null;
    }
    
    // Checkouts - server can write without auth
    match /checkouts/{checkoutId} {
      allow read: if request.auth != null || request.auth == null;
      allow write: if request.auth != null || request.auth == null;
    }
    
    // Sales - server can write without auth
    match /sales/{saleId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null || request.auth == null;
    }
    
    // Counters
    match /_counters/{counterId} {
      allow read, write: if request.auth == null;
    }
  }
}
